19  /**
20   * Runs tests with varnishtest.
21   *
22   * <pre>{@code <project xmlns:varnishtest="antlib:com.zenika.varnishtest.ant"
23   *     name="minimal-configuration" basedir="." default="test">
24   * 
25   *     <target name="test" description="Run varnishtest">
26   *         <varnishtest:run>
27   *             <fileset dir="src/test/varnish">
28   *                 <include name="**.vtc"/>
29   *             </fileset>
30   *         </varnishtest:run>
31   *     </target>
32   * 
33   * </project>}</pre>
34   *
35   * @author Dridi Boukelmoune
36   * @ant.task
37   */
38  public class RunTask extends Task {
39  
40  	/**
41  	 * Default test execution timeout in seconds.
42  	 */
43  	public static final int DEFAULT_TIMEOUT = 20;
44  
45  	private final CommandLineBuilder builder = new CommandLineBuilder();
46  	private final List<ResourceCollection> collections = new ArrayList<ResourceCollection>();
47  	private final LogOutputHandler handler = new LogOutputHandler(this);
48  	
49  	private File workingDirectory = null;
50  	private File reportsDirectory = null;
51  	private int timeout = DEFAULT_TIMEOUT;
52  	private int testCaseCounter = 1;
53  
54  	/**
55  	 * Initialize default values.
56  	 * @since 0.2
57  	 */
58  	@Override
59  	public void init() {
60  		workingDirectory = getProject().getBaseDir();
61  	}
62  
63  	/**
64  	 * Run the varnish test cases (vtc).
65  	 */
66  	@Override
67  	public void execute() {
68  		createReportsDirectory();
69  
70  		for (ResourceCollection collection : collections) {
71  			@SuppressWarnings("unchecked")
72  			Iterator<Resource> iterator = collection.iterator();
73  			while (iterator.hasNext()) {
74  				runTestCase(iterator.next(), testCaseCounter);
75  				testCaseCounter++;
76  			}
77  		}
78  	}
79  
80  	private void createReportsDirectory() {
81  		if (reportsDirectory == null) {
82  			return;
83  		}
84  
85  		reportsDirectory.mkdirs();
86  
87  		if ( ! reportsDirectory.exists() ) {
88  			throw new BuildException("Varnishtest reports directory not created : " + reportsDirectory);
89  		}
90  	}
91  
92  	private void writeReport(VarnishtestReport report, int number) {
93  		if (reportsDirectory == null) {
94  			return;
95  		}
96  
97  		try {
98  			report.write(reportsDirectory, number);
99  		}
100 		catch (IOException e) {
101 			throw new BuildException("Could not write varnishtest's report", e);
102 		}
103 	}
104 
105 	private void runTestCase(Resource resource, int testCaseCounter) {
106 		VarnishtestRunner runner = new VarnishtestRunner(builder, workingDirectory);
107 		try {
108 			VarnishtestReport report = runner.runTestCase(new File(resource.toString()), handler, timeout);
109 			writeReport(report, testCaseCounter);
110 		}
111 		catch (VarnishtestException e) {
112 			writeReport(e.getReport(), testCaseCounter);
113 			throw new BuildException("Test failure : " + resource, e);
114 		}
115 		catch (IOException e) {
116 			throw new BuildException("An error occured", e);
117 		}
118 	}

131 		}
