9   /**
10   * This class allows to build a proper command line for varnishtest execution.
11   * The test case (vtc) file that should be part of the command line is expected
12   * in the {@link VarnishtestRunner} class.
13   * 
14   * <p>
15   * Varnishtest accepts macros, but the following name cannot be set :
16   * <ul>
17   *  <li>varnishd</li>
18   * </ul>
19   * Macro names are validated against this regex : {@value #MACRO_NAME_REGEX}
20   * </p>
21   * 
22   * @author Dridi Boukelmoune
23   * @see VarnishtestRunner#VarnishtestRunner(CommandLineBuilder)
24   * @see VarnishtestRunner#runTestCase(java.io.File)
25   */
26  public class CommandLineBuilder {
27  	
28  	public  static final String MACRO_NAME_REGEX = "\\w+(\\.\\w+)*";
29  	private static final Pattern MACRO_NAME_PATTERN = Pattern.compile(MACRO_NAME_REGEX);
30  	private static final List<String> RESERVED_MACROS = Arrays.asList("varnishd", "pwd", "tmpdir", "bad_ip");
31

33  	private String varnishdCommand = "varnishd";

111 	/**
112 	 * Returns the value of a macro.
113 	 * @param name the name of the macro
114 	 * @return the value of the macro
115 	 * @throws NullPointerException if {@code name} is {@code null}
116 	 * @throws IllegalArgumentException if {@code name} does not exist
117 	 */
118 	public String getMacro(String name) {
119 		if (name == null) {
120 			throw new NullPointerException("Null name");
121 		}
122 		if ( ! macros.containsKey(name) ) {
123 			throw new IllegalArgumentException("Unknown macro : " + name);
124 		}
125 		return macros.get(name);
126 	}
127 	
128 	/**
129 	 * Sets a collection of macros.
130 	 * @param macros the macros to set
131 	 * @return {@code this} builder
132 	 * @throws NullPointerException if {@code macros} is {@code null}
133 	 * @throws NullPointerException if {@code macros} contains a {@code null} name or value
134 	 * @throws IllegalArgumentException if {@code macros} contains a reserved name
135 	 * @throws IllegalArgumentException if {@code macros} contains an invalid name
136 	 */
137 	public CommandLineBuilder setMacros(Map<String, String> macros) {
138 		if (macros == null) {
139 			throw new NullPointerException("Null macros");
140 		}
141 		for (Entry<String, String> entry : macros.entrySet()) {
142 			setMacro(entry.getKey(), entry.getValue());
143 		}
144 		return this;
145 	}
146 	
147 	/**
148 	 * Sets the value of a macro.
149 	 * @param name the name of the macro
150 	 * @param value the value of the macro
151 	 * @return {@code this} builder
152 	 * @throws NullPointerException if {@code name} or {@code value} is {@code null}
153 	 * @throws IllegalArgumentException if {@code name} is reserved
154 	 * @throws IllegalArgumentException if {@code name} is not valid
155 	 */
156 	public CommandLineBuilder setMacro(String name, String value) {
157 		if (name == null || value == null) {
158 			throw new NullPointerException("Null name or value");
159 		}
160 		if ( RESERVED_MACROS.contains(name) ) {
161 			throw new IllegalArgumentException("The macro `" + name + "' is not allowed");
162 		}
163 		if ( MACRO_NAME_PATTERN.matcher(name).matches() ) {
164 			macros.put(name, value);
165 			return this;
166 		}
167 		throw new IllegalArgumentException("Invalid macro name : " + name);
168 	}
169 	
170 	/**
171 	 * Removes all macros previously set.
172 	 * @return {@code this} builder
173 	 */
174 	public CommandLineBuilder clearMacros() {
175 		macros.clear();
176 		return this;
177 	}

179 	/**
180 	 * Create a commons-exec {@link CommandLine}.
181 	 * @return the command line
182 	 */
183 	CommandLine toCommandLine() {
184 		CommandLine commandLine = new CommandLine(varnishtestCommand);
185 		commandLine.addArgument("-v"); // we need the verbose flag for the report
186 		
187 		commandLine.addArgument("-Dvarnishd=" + varnishdCommand, false);
188 		
189 		for (Map.Entry<String, String> macro : macros.entrySet()) {
190 			commandLine.addArgument("-D" + macro.getKey() + "=" + macro.getValue(), false);
191 		}
192 
193 		return commandLine;
194 	}
